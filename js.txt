// Run this on a YouTube live stream watch page (https://www.youtube.com/watch?v=...)
// Open DevTools (F12) → Console tab → paste → Enter.
(() => {
  const player = window.ytInitialPlayerResponse || {};
  const initial = window.ytInitialData || {};

  console.log('=== videoDetails ===');
  console.log({
    isLive: player.videoDetails?.isLive,
    isLiveContent: player.videoDetails?.isLiveContent,
    viewCount: player.videoDetails?.viewCount,
    shortViewCount: player.videoDetails?.shortViewCount,
  });

  console.log('=== microformat ===');
  const mf = player.microformat?.playerMicroformatRenderer || {};
  console.log({
    viewCount: mf.viewCount,
    liveBroadcastDetails: mf.liveBroadcastDetails,
  });

  console.log('=== videoPrimaryInfoRenderer (concurrent viewers UI) ===');
  const contents = initial.contents?.twoColumnWatchNextResults?.results?.results?.contents;
  if (contents) {
    for (const item of contents) {
      if (item.videoPrimaryInfoRenderer) {
        const vcr = item.videoPrimaryInfoRenderer.viewCount?.videoViewCountRenderer;
        console.log({
          simpleText: vcr?.viewCount?.simpleText,
          runs: vcr?.viewCount?.runs?.map(r => r.text).join(''),
          shortViewCount: vcr?.shortViewCount?.simpleText,
          isLive: vcr?.isLive,
        });
      }
    }
  }

  console.log('=== HTML search for "watching" / "assistindo" ===');
  const html = document.documentElement.outerHTML;
  const matches = html.match(/[\d.,\s]+\s*(?:watching now|assistindo agora|assistindo)/gi);
  console.log(matches?.slice(0, 5));
})();
